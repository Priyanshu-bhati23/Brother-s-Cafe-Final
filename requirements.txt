- react-dom@^18.2.0
- framer-motion@^10.16.4
- lucide-react@^0.294.0

- vite@^5.0.8
- @vitejs/plugin-react@^4.2.1
- tailwindcss@^3.3.6
- autoprefixer@^10.4.16
- postcss@^8.4.32

npm run dev
